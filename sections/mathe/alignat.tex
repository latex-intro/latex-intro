Betrachten wir nun folgendes Beispiel: 

\begin{codeblock}
\begin{verbatim}
\begin{alignat}{4}
a + (3 + 1) &= a + 4          &=&a + (2 + 2)\\
            &= a + (0 + 4)    &=&a + (-1 + 5)\\
b + 102     &= b + 100        &=&b + (49 + 51)\\
            &= b + 1000 - 900 &=&b + (49 + 51)
\end{alignat}
\end{verbatim}
\end{codeblock}
\end{frame}

\begin{frame}[fragile]{Die \umg{alignat}-Umgebung}
\framesubtitle{Benutzerdefinierte Nummerierung}
Benutzerdefinierte Nummerierung durch die Befehle \cmd{notag} und \cmd{tag}\marg{Text}.

\begin{columns}
\begin{column}{0.5\textwidth}
\begin{codeblock}
\begin{tiny}
\begin{verbatim}
\begin{alignat}{2}
 x^2 &=1 \\
 a^2 + b^2 &= c^2 \notag\\ 
 i\hbar \frac{\partial}{\partial t}|\psi(t)
 \rangle&=\mathbb{H}|\psi(t)\rangle \tag{S}
\end{alignat}
\end{verbatim}
\end{tiny}
\end{codeblock}
\end{column}
\begin{column}{0.45\textwidth}

\end{column}
\end{columns}
